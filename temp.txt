384:    def _detect_semantic_gaps(self, clusters: Dict) -> List[PhantomCandidate]:
385:        """
386:        Детекция семантических лакун (алгоритм по спецификации)
387:        
388:        Расчёт центроида: μC = (1/|C|)Σv
389:        Семантический разрыв: ΔC = min(1 - cos(μC, v))
390:        Адаптивный порог: τ = τ_base * (1 + σ)
391:        """
392:        candidates = []
393:        base_threshold = self.config.get("base_threshold", 0.30)
394:        dedup_radius = self.config.get("dedup_radius", 0.15)
395:        
396:        for cluster_id, node_ids in clusters.items():
397:            if len(node_ids) < 3:
398:                continue
399:            
400:            try:
401:                nodes_data = self._get_cluster_nodes_data(node_ids)
402:                if not nodes_data or len(nodes_data) < 3:
403:                    continue
404:                
405:                embeddings = [n.get('embedding') for n in nodes_data if n.get('embedding')]
406:                if not embeddings:
407:                    continue
408:                
409:                embeddings = [np.array(e) for e in embeddings if len(e) > 0]
410:                if not embeddings:
411:                    continue
412:                
413:                centroid = np.mean(embeddings, axis=0)
414:                centroid = centroid / (np.linalg.norm(centroid) + 1e-8)
415:                
416:                gaps = []
417:                for emb in embeddings:
418:                    cos_sim = np.dot(centroid, emb) / (np.linalg.norm(emb) + 1e-8)
419:                    gap = 1.0 - cos_sim
420:                    gaps.append(gap)
421:                
422:                variance = float(np.std(gaps))
423:                semantic_gap = float(min(gaps))
424:                
425:                variance_k = self.config.get("variance_k", 2.0)
426:                threshold = base_threshold * (1 + variance_k * variance)
427:                
428:                if semantic_gap > threshold:
429:                    if not self._is_duplicate(centroid, dedup_radius):
430:                        candidate = PhantomCandidate(
431:                            id=f"phantom_{int(time.time() * 1000)}_{hash(cluster_id) % 10000}",
432:                            cluster_id=cluster_id,
433:                            centroid=centroid.tolist(),
434:                            nodes=nodes_data,
435:                            variance=variance,
436:                            semantic_gap=semantic_gap
437:                        )
438:                        candidates.append(candidate)
439:                        
440:            except Exception as e:
441:                logger.warning(f"Ошибка анализа кластера {cluster_id}: {e}")
442:        
443:        self._metrics["phantom_detection_rate"] = len(candidates) / max(len(clusters), 1)
444:        
445:        return candidates

468:    def _is_duplicate(self, centroid: np.ndarray, radius: float) -> bool:
469:        """Проверка на дубликат среди существующих концептов"""
470:        for candidate in self._candidates.values():
471:            if candidate.status in ["confirmed", "stable"] and candidate.centroid:
472:                existing = np.array(candidate.centroid)
473:                cos_sim = np.dot(centroid, existing) / (np.linalg.norm(existing) + 1e-8)

475:                    return True

588:    def _calculate_lexical_entropy(self, text: str) -> float:
589:        """Расчёт лексической энтропии для адаптации параметров"""
590:        if not text:
591:            return 0.0
592:        
593:        words = text.lower().split()
594:        if len(words) < 2:
595:            return 0.0
596:        
597:        word_freq = {}
598:        for word in words:
599:            word_freq[word] = word_freq.get(word, 0) + 1
600:        
601:        total = len(words)
602:        entropy = 0.0
603:        for freq in word_freq.values():
604:            p = freq / total
605:            if p > 0:
606:                entropy -= p * (p ** 0.5)
607:        
608:        return entropy
